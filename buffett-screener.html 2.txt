// ── INTRINSIC VALUE ─────────────────────────────────────────────────

function calcGrahamNumber(eps, bvps) {
  if (eps === null || bvps === null || eps <= 0 || bvps <= 0) return null;
  return Math.sqrt(22.5 * eps * bvps);
}

// Graham's intrinsic value formula: V = EPS × (8.5 + 2g) × 4.4 / Y
// g = expected growth rate (%), Y = current AAA/10yr yield (%)
function calcGrahamIV(eps, epsGrowth, riskFreeRate) {
  if (eps === null || epsGrowth === null) return null;
  const g = epsGrowth;
  const Y = riskFreeRate !== null && riskFreeRate > 0 ? riskFreeRate : 4.5;
  const iv = eps * (8.5 + 2 * g) * (4.4 / Y);
  return iv > 0 ? iv : null;
}

function score() {
  // Read inputs
  const price     = getVal('price');
  const eps       = getVal('eps');
  const bvps      = getVal('bvps');
  const pe        = getVal('pe');
  const pb        = getVal('pb');
  const fcfY      = getVal('fcfYield');
  const rfr       = getVal('riskFreeRate');
  const roe       = getVal('roe');
  const margin    = getVal('margin');
  const epsG      = getVal('epsGrowth');
  const revG      = getVal('revGrowth');
  const de        = getVal('de');
  const cr        = getVal('cr');

  const moatChecks = ['chk1','chk2','chk3','chk4','chk5','chk6']
    .filter(id => document.getElementById(id).checked).length;
  const moatPts = moatChecks * 2;

  // Individual scores
  const peS      = scorePE(pe);
  const pbS      = scorePB(pb);
  const fcfS     = scoreFCF(fcfY);
  const roeS     = scoreROE(roe);
  const marginS  = scoreMargin(margin);
  const epsgS    = scoreEPSG(epsG);
  const revgS    = scoreRevG(revG);
  const deS      = scoreDE(de);
  const crS      = scoreCR(cr);

  // Update individual metric bars (inline)
  updateMetricBar('pe',     peS,     MAX.pe);
  updateMetricBar('pb',     pbS,     MAX.pb);
  updateMetricBar('fcf',    fcfS,    MAX.fcf);
  updateMetricBar('roe',    roeS,    MAX.roe);
  updateMetricBar('margin', marginS, MAX.margin);
  updateMetricBar('epsg',   epsgS,   MAX.epsg);
  updateMetricBar('revg',   revgS,   MAX.revg);
  updateMetricBar('de',     deS,     MAX.de);
  updateMetricBar('cr',     crS,     MAX.cr);

  // Breakdown panel
  setBreakdown('pe',     peS,    MAX.pe);
  setBreakdown('pb',     pbS,    MAX.pb);
  setBreakdown('fcf',    fcfS,   MAX.fcf);
  setBreakdown('roe',    roeS,   MAX.roe);
  setBreakdown('margin', marginS,MAX.margin);
  setBreakdown('epsg',   epsgS,  MAX.epsg);
  setBreakdown('revg',   revgS,  MAX.revg);
  setBreakdown('de',     deS,    MAX.de);
  setBreakdown('cr',     crS,    MAX.cr);
  setBreakdown('moat',   moatPts,MAX.moat);

  // Count how many metrics were entered
  const scores = [peS, pbS, fcfS, roeS, marginS, epsgS, revgS, deS, crS];
  const filled = scores.filter(s => s !== null);

  if (filled.length === 0 && moatChecks === 0) {
    resetResults();
    return;
  }

  // Scale quantitative total to 88 pts, add moat
  const maxQuant = [MAX.pe, MAX.pb, MAX.fcf, MAX.roe, MAX.margin, MAX.epsg, MAX.revg, MAX.de, MAX.cr];
  let earnedQuant = 0, maxQuantFilled = 0;
  const allScores = [peS, pbS, fcfS, roeS, marginS, epsgS, revgS, deS, crS];
  const allMax    = [MAX.pe, MAX.pb, MAX.fcf, MAX.roe, MAX.margin, MAX.epsg, MAX.revg, MAX.de, MAX.cr];
  for (let i = 0; i < allScores.length; i++) {
    if (allScores[i] !== null) {
      earnedQuant  += allScores[i];
      maxQuantFilled += allMax[i];
    }
  }

  // Normalize: scale earned to what 88 pts would be if all were filled
  const totalQuant = maxQuantFilled > 0 ? Math.round((earnedQuant / maxQuantFilled) * 88) : 0;
  const totalScore = Math.min(100, totalQuant + moatPts);

  // Ring
  const circ = 263.9;
  const offset = circ - (totalScore / 100) * circ;
  const ring = document.getElementById('ringFill');
  ring.style.strokeDashoffset = offset;
  const ringColor = totalScore >= 70 ? '#00C805'
                  : totalScore >= 55 ? '#fbbf24'
                  : '#FF5000';
  ring.style.stroke = ringColor;

  document.getElementById('scoreNum').textContent = totalScore;
  document.getElementById('scoreNum').style.color = ringColor;

  // Recommendation
  let rec, recColor, recBg, desc, dotColor;
  if (totalScore >= 80) {
    rec = 'Strong Buy'; recColor = '#fff'; recBg = 'rgba(0,200,5,0.2)';
    desc = 'This stock checks most of Buffett\'s boxes — cheap, profitable, financially sound, and likely has a moat. Consider a full position with a 25%+ margin of safety.';
    dotColor = '#00C805';
  } else if (totalScore >= 65) {
    rec = 'Buy'; recColor = '#00C805'; recBg = 'rgba(0,200,5,0.10)';
    desc = 'Solid value opportunity. A few metrics are below ideal Buffett thresholds, but the overall picture is attractive. Consider a starter or half position.';
    dotColor = '#00C805';
  } else if (totalScore >= 50) {
    rec = 'Watch / Hold'; recColor = '#fbbf24'; recBg = 'rgba(251,191,36,0.12)';
    desc = 'Fair price, not a great deal. Buffett would likely wait for a better entry point or a catalyst that improves the financials before committing.';
    dotColor = '#fbbf24';
  } else if (totalScore >= 35) {
    rec = 'Avoid'; recColor = '#FF5000'; recBg = 'rgba(255,80,0,0.12)';
    desc = 'Overpriced or financially weak relative to Buffett\'s standards. Consider only if the business has exceptional qualitative moat not captured in the numbers.';
    dotColor = '#FF5000';
  } else {
    rec = 'Strong Avoid'; recColor = '#fff'; recBg = 'rgba(255,80,0,0.22)';
    desc = 'This stock would not meet Buffett\'s criteria on any meaningful level. Speculative at best. Significant downside risk relative to intrinsic value.';
    dotColor = '#FF5000';
  }

  document.getElementById('resDot').style.background = dotColor;
  const ticker = document.getElementById('ticker').value.trim().toUpperCase() || 'Stock';
  const name   = document.getElementById('companyName').value.trim() || ticker;
  document.getElementById('resTitle').textContent = name || 'Result';
  const badge = document.getElementById('recBadge');
  badge.textContent = rec;
  badge.style.background = recBg;
  badge.style.color = recColor;
  document.getElementById('recDesc').textContent = desc;

  // Intrinsic Value
  const gn  = calcGrahamNumber(eps, bvps);
  const giv = calcGrahamIV(eps, epsG, rfr);

  // Graham Number
  const gnEl  = document.getElementById('grahamVal');
  const gnSub = document.getElementById('grahamSub');
  const gnBar = document.getElementById('grahamBar');
  if (gn !== null) {
    gnEl.textContent = fmt$(gn);
    gnEl.style.color = price !== null ? (price < gn ? 'var(--good)' : 'var(--bad)') : '#fff';
    if (price !== null) {
      const mos = ((gn - price) / gn) * 100;
      gnSub.textContent = price < gn
        ? `${mos.toFixed(0)}% discount to fair value`
        : `${Math.abs(mos).toFixed(0)}% premium — overvalued`;
      const barW = Math.min(100, Math.max(0, (price / gn) * 100));
      gnBar.style.width = barW + '%';
      gnBar.style.background = price < gn ? 'var(--good)' : 'var(--bad)';
    } else {
      gnSub.textContent = 'Enter price to compare';
      gnBar.style.width = '50%';
      gnBar.style.background = 'var(--muted2)';
    }
  } else {
    gnEl.textContent = '—';
    gnEl.style.color = '#fff';
    gnSub.textContent = '√(22.5 × EPS × BVPS)';
    gnBar.style.width = '0%';
  }

  // Graham IV
  const givEl  = document.getElementById('grahamIVVal');
  const givSub = document.getElementById('grahamIVSub');
  const givBar = document.getElementById('grahamIVBar');
  if (giv !== null) {
    givEl.textContent = fmt$(giv);
    givEl.style.color = price !== null ? (price < giv ? 'var(--good)' : 'var(--bad)') : '#fff';
    if (price !== null) {
      const mos = ((giv - price) / giv) * 100;
      givSub.textContent = price < giv
        ? `${mos.toFixed(0)}% discount to IV`
        : `${Math.abs(mos).toFixed(0)}% premium to IV`;
      const barW = Math.min(100, Math.max(0, (price / giv) * 100));
      givBar.style.width = barW + '%';
      givBar.style.background = price < giv ? 'var(--good)' : 'var(--bad)';
    } else {
      givSub.textContent = 'Enter price to compare';
      givBar.style.width = '50%';
      givBar.style.background = 'var(--muted2)';
    }
  } else {
    givEl.textContent = '—';
    givEl.style.color = '#fff';
    givSub.textContent = 'EPS × (8.5 + 2g) × 4.4 / Y';
    givBar.style.width = '0%';
  }

  // Margin of Safety vs Graham Number
  const mosEl  = document.getElementById('mosVal');
  const mosSub = document.getElementById('mosSub');
  const mosBar = document.getElementById('mosBar');
  if (gn !== null && price !== null) {
    const mos = ((gn - price) / gn) * 100;
    mosEl.textContent = mos.toFixed(1) + '%';
    mosEl.style.color = mos >= 25 ? 'var(--good)'
                       : mos >= 0  ? 'var(--warn)'
                       : 'var(--bad)';
    mosSub.textContent = mos >= 25 ? 'Excellent margin of safety — Buffett\'s preferred ≥25% buffer'
                        : mos >= 0  ? 'Thin margin of safety — wait for a better entry'
                        : 'Stock is trading above Graham Number — no margin of safety';
    const barW = Math.min(100, Math.max(0, (mos / 50) * 100)); // 50% MOS = full bar
    mosBar.style.width = barW + '%';
    mosBar.style.background = mos >= 25 ? 'var(--good)' : mos >= 0 ? 'var(--warn)' : 'var(--bad)';
  } else {
    mosEl.textContent = '—';
    mosEl.style.color = '#fff';
    mosSub.textContent = 'Enter price, EPS, and book value to compute';
    mosBar.style.width = '0%';
  }

  // Quote
  const q = QUOTES.find(q => totalScore >= q.min) || QUOTES[QUOTES.length - 1];
  document.getElementById('quoteText').textContent = q.text;
}

// Update per-metric inline bar (inside input card)
function updateMetricBar(key, pts, max) {
  const bar = document.getElementById('bar-' + key);
  const ptsEl = document.getElementById('pts-' + key);
  if (!bar) return;
  if (pts === null) {
    bar.style.width = '0%';
    bar.style.background = 'rgba(255,255,255,0.08)';
    if (ptsEl) ptsEl.textContent = '— pts';
    return;
  }
  const pct = (pts / max) * 100;
  const color = pts >= max * 0.7 ? 'var(--good)' : pts >= max * 0.35 ? 'var(--warn)' : 'var(--bad)';
  bar.style.width = pct + '%';
  bar.style.background = color;
  if (ptsEl) {
    ptsEl.textContent = pts + ' / ' + max + ' pts';
    ptsEl.style.color = color;
  }
}

  document.getElementById('scoreNum').textContent = '—';
  document.getElementById('scoreNum').style.color = '#fff';
  document.getElementById('resDot').style.background = 'rgba(255,255,255,0.15)';

  const badge = document.getElementById('recBadge');
  badge.textContent = 'Enter Data';
  badge.style.background = 'rgba(255,255,255,0.05)';

  document.getElementById('recDesc').textContent = 'Fill in the financial metrics on the left to generate a Buffett-style buy recommendation.';
  document.getElementById('quoteText').textContent = '"Price is what you pay. Value is what you get."';
}
